import React, { useState, useEffect, useMemo } from 'react';
import { 
  TrendingUp, 
  TrendingDown, 
  Search, 
  BarChart3, 
  PieChart, 
  ArrowUpRight, 
  ArrowDownRight,
  Building2,
  Activity,
  Globe,
  RefreshCw,
  Info,
  Download,
  Calendar,
  Bell,
  Trash2,
  Plus,
  BellRing
} from 'lucide-react';
import { 
  LineChart, 
  Line, 
  XAxis, 
  YAxis, 
  CartesianGrid, 
  Tooltip, 
  ResponsiveContainer,
  AreaChart,
  Area
} from 'recharts';
import { motion, AnimatePresence } from 'motion/react';
import { cn } from './lib/utils';
import { fetchMarketOverview, fetchMarketNews } from './services/geminiService';
import { StockData, CASABLANCA_COMPANIES, Alert, NewsItem } from './types';

export default function App() {
  const [stocks, setStocks] = useState<StockData[]>([]);
  const [news, setNews] = useState<NewsItem[]>([]);
  const [loading, setLoading] = useState(true);
  const [loadingNews, setLoadingNews] = useState(false);
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedStock, setSelectedStock] = useState<StockData | null>(null);
  const [lastUpdated, setLastUpdated] = useState<Date>(new Date());
  const [alerts, setAlerts] = useState<Alert[]>([]);
  const [newAlertPrice, setNewAlertPrice] = useState<string>('');
  const [newAlertCondition, setNewAlertCondition] = useState<'above' | 'below'>('above');

  // Load alerts from localStorage
  useEffect(() => {
    const savedAlerts = localStorage.getItem('casafinance_alerts');
    if (savedAlerts) {
      setAlerts(JSON.parse(savedAlerts));
    }
  }, []);

  // Save alerts to localStorage
  useEffect(() => {
    localStorage.setItem('casafinance_alerts', JSON.stringify(alerts));
  }, [alerts]);

  const loadData = async () => {
    setLoading(true);
    setLoadingNews(true);
    
    // Fetch stocks and news in parallel
    const [stockData, newsData] = await Promise.all([
      fetchMarketOverview(),
      fetchMarketNews()
    ]);

    if (stockData && stockData.length > 0) {
      setStocks(stockData.map(s => ({
        ...s,
        logo: s.logo || `https://picsum.photos/seed/${s.symbol}/100/100`
      })));
    } else {
      // Fallback/Mock data if API fails or returns empty
      const mockData: StockData[] = CASABLANCA_COMPANIES.map(name => ({
        symbol: name.substring(0, 4).toUpperCase(),
        name,
        price: Math.random() * 1000 + 100,
        change: (Math.random() - 0.5) * 20,
        changePercent: (Math.random() - 0.5) * 5,
        volume: Math.floor(Math.random() * 50000),
        marketCap: (Math.random() * 100).toFixed(2) + " Mrd MAD",
        sector: name.includes("BANK") || name.includes("CREDIT") ? "Banques" : "Industrie",
        logo: `https://picsum.photos/seed/${name}/100/100`,
        history: Array.from({ length: 7 }, (_, i) => ({
          date: `Day ${i + 1}`,
          price: Math.random() * 1000 + 100
        }))
      }));
      setStocks(mockData);
    }

    if (newsData && newsData.length > 0) {
      setNews(newsData);
    } else {
      // Fallback news
      setNews([
        { title: "Résultats annuels: Le secteur bancaire en tête", time: "Il y a 2h", sentiment: "positive", summary: "Les banques marocaines affichent des bénéfices records cette année." },
        { title: "MASI: Une clôture en légère hausse ce vendredi", time: "Il y a 5h", sentiment: "neutral", summary: "Le marché reste stable malgré une volatilité matinale." },
        { title: "Nouveaux investissements dans le secteur minier", time: "Il y a 1j", sentiment: "positive", summary: "Managem annonce un nouveau projet d'expansion majeur." }
      ]);
    }

    setLastUpdated(new Date());
    setLoading(false);
    setLoadingNews(false);
  };

  const exportToCSV = () => {
    const headers = ["Symbole", "Nom", "Prix (MAD)", "Variation %", "Volume", "Cap. Boursière", "Secteur"];
    const rows = stocks.map(s => [
      s.symbol,
      s.name,
      s.price.toFixed(2),
      s.changePercent.toFixed(2),
      s.volume,
      s.marketCap,
      s.sector
    ]);

    const csvContent = [
      headers.join(","),
      ...rows.map(r => r.join(","))
    ].join("\n");

    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement("a");
    const url = URL.createObjectURL(blob);
    link.setAttribute("href", url);
    link.setAttribute("download", `CasaFinance_Export_${new Date().toLocaleDateString()}.csv`);
    link.style.visibility = 'hidden';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const addAlert = () => {
    if (!selectedStock || !newAlertPrice) return;
    
    const alert: Alert = {
      id: Math.random().toString(36).substr(2, 9),
      symbol: selectedStock.symbol,
      name: selectedStock.name,
      targetPrice: parseFloat(newAlertPrice),
      condition: newAlertCondition,
      isActive: true,
      createdAt: new Date().toISOString()
    };
    
    setAlerts([...alerts, alert]);
    setNewAlertPrice('');
  };

  const removeAlert = (id: string) => {
    setAlerts(alerts.filter(a => a.id !== id));
  };

  const checkAlerts = (currentStocks: StockData[]) => {
    alerts.forEach(alert => {
      const stock = currentStocks.find(s => s.symbol === alert.symbol);
      if (stock && alert.isActive) {
        const triggered = alert.condition === 'above' 
          ? stock.price >= alert.targetPrice 
          : stock.price <= alert.targetPrice;
        
        if (triggered) {
          // In a real app we might send a notification
          console.log(`ALERT TRIGGERED: ${alert.name} is ${alert.condition} ${alert.targetPrice}`);
        }
      }
    });
  };

  useEffect(() => {
    if (stocks.length > 0) {
      checkAlerts(stocks);
    }
  }, [stocks]);

  useEffect(() => {
    loadData();
  }, []);

  const filteredStocks = useMemo(() => {
    return stocks.filter(s => 
      s.name.toLowerCase().includes(searchTerm.toLowerCase()) || 
      s.symbol.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }, [stocks, searchTerm]);

  const topGainers = useMemo(() => {
    return [...stocks].sort((a, b) => b.changePercent - a.changePercent).slice(0, 4);
  }, [stocks]);

  const topFlops = useMemo(() => {
    return [...stocks].sort((a, b) => a.changePercent - b.changePercent).slice(0, 4);
  }, [stocks]);

  const marketStats = useMemo(() => {
    const up = stocks.filter(s => s.changePercent > 0).length;
    const down = stocks.filter(s => s.changePercent < 0).length;
    const total = stocks.length;
    return { up, down, total };
  }, [stocks]);

  const sectors = [
    { name: "Banques", icon: "🏦", change: 1.2, color: "bg-blue-500" },
    { name: "Immobilier", icon: "🏠", change: -0.5, color: "bg-orange-500" },
    { name: "Télécoms", icon: "📱", change: 0.8, color: "bg-purple-500" },
    { name: "Mines", icon: "⛏️", change: 2.1, color: "bg-amber-500" },
    { name: "Assurances", icon: "🛡️", change: 0.3, color: "bg-emerald-500" },
  ];

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900 selection:bg-blue-100">
      {/* Header */}
      <header className="sticky top-0 z-50 glass border-b border-slate-200 px-6 py-4">
        <div className="max-w-7xl mx-auto flex flex-col md:flex-row md:items-center justify-between gap-4">
          <div className="flex items-center gap-3">
            <div className="bg-blue-600 p-2 rounded-xl shadow-lg shadow-blue-200">
              <Activity className="text-white w-6 h-6" />
            </div>
            <div>
              <h1 className="text-xl font-bold tracking-tight">CasaFinance</h1>
              <p className="text-xs text-slate-500 font-medium uppercase tracking-wider">Bourse de Casablanca (MASI)</p>
            </div>
          </div>

          <div className="flex items-center gap-4">
            <button 
              onClick={exportToCSV}
              className="hidden md:flex items-center gap-2 px-4 py-2 bg-white border border-slate-200 rounded-full text-sm font-bold text-slate-600 hover:bg-slate-50 transition-all"
            >
              <Download className="w-4 h-4" />
              Exporter CSV
            </button>
            <div className="relative group">
              <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400 group-focus-within:text-blue-500 transition-colors" />
              <input 
                type="text" 
                placeholder="Rechercher une société..." 
                className="pl-10 pr-4 py-2 bg-white border border-slate-200 rounded-full text-sm focus:outline-none focus:ring-2 focus:ring-blue-500/20 focus:border-blue-500 transition-all w-full md:w-64"
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
              />
            </div>
            <button 
              onClick={loadData}
              disabled={loading}
              className="p-2 hover:bg-slate-100 rounded-full transition-colors disabled:opacity-50"
            >
              <RefreshCw className={cn("w-5 h-5 text-slate-600", loading && "animate-spin")} />
            </button>
          </div>
        </div>
      </header>

      <main className="max-w-7xl mx-auto px-6 py-8 space-y-8">
        {/* Hero Banner with Bright Casa Finance City Vibe */}
        <section className="relative h-[400px] rounded-[40px] overflow-hidden shadow-2xl group border border-white">
          <img 
            src="https://images.unsplash.com/photo-1486406146926-c627a92ad1ab?auto=format&fit=crop&q=80&w=2000" 
            alt="Casa Finance City Skyline" 
            className="absolute inset-0 w-full h-full object-cover group-hover:scale-105 transition-transform duration-1000"
            referrerPolicy="no-referrer"
          />
          <div className="absolute inset-0 bg-gradient-to-r from-white/95 via-white/60 to-transparent flex flex-col justify-center px-12">
            <motion.div
              initial={{ opacity: 0, x: -30 }}
              animate={{ opacity: 1, x: 0 }}
              transition={{ duration: 0.8 }}
            >
              <div className="flex items-center gap-2 mb-4">
                <span className="px-3 py-1 bg-blue-600 text-white text-[10px] font-black uppercase tracking-widest rounded-full">Direct Live</span>
                <span className="text-blue-900/60 text-[10px] font-bold uppercase tracking-widest">Casablanca Finance City</span>
              </div>
              <h2 className="text-5xl md:text-7xl font-black text-slate-900 mb-6 tracking-tighter leading-[0.85]">
                L'Élite de la <br />
                <span className="text-blue-600">Finance Marocaine</span>
              </h2>
              <p className="text-slate-700 text-lg max-w-xl font-semibold leading-relaxed mb-8">
                Analyse en temps réel des 75 sociétés cotées à la Bourse de Casablanca. Propulsé par l'IA pour des décisions d'investissement éclairées.
              </p>
              <div className="flex gap-4">
                <button className="px-8 py-4 bg-blue-600 text-white rounded-2xl font-black text-sm hover:bg-blue-700 transition-all shadow-xl shadow-blue-500/20">
                  Explorer le MASI
                </button>
                <button className="px-8 py-4 bg-white text-blue-600 border border-blue-100 rounded-2xl font-black text-sm hover:bg-blue-50 transition-all shadow-sm">
                  Alertes de Prix
                </button>
              </div>
            </motion.div>
          </div>
        </section>

        {/* Market Overview Cards */}
        <section className="grid grid-cols-1 md:grid-cols-3 lg:grid-cols-4 gap-6">
          <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100 flex flex-col justify-between">
            <div className="flex items-center justify-between mb-4">
              <span className="text-sm font-semibold text-slate-500 uppercase tracking-wider">État du Marché</span>
              <Globe className="w-5 h-5 text-blue-500" />
            </div>
            <div className="space-y-2">
              <div className="flex items-end gap-2">
                <span className="text-3xl font-bold">MASI</span>
                <span className="text-sm font-medium text-emerald-500 mb-1 flex items-center">
                  <ArrowUpRight className="w-4 h-4" /> +0.42%
                </span>
              </div>
              <p className="text-xs text-slate-400">Dernière mise à jour: {lastUpdated.toLocaleTimeString()}</p>
            </div>
          </div>

          <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100">
            <div className="flex items-center justify-between mb-4">
              <span className="text-sm font-semibold text-slate-500 uppercase tracking-wider">Alertes Actives</span>
              <Bell className="w-5 h-5 text-amber-500" />
            </div>
            <div className="flex items-center gap-4">
              <div className="flex-1">
                <div className="flex items-center gap-2">
                  <span className="text-3xl font-bold">{alerts.length}</span>
                  <span className="text-xs font-medium text-slate-400">Surveillances en cours</span>
                </div>
                <div className="flex gap-1 mt-2">
                  {alerts.slice(0, 3).map(a => (
                    <div key={a.id} className="w-2 h-2 rounded-full bg-amber-400" />
                  ))}
                  {alerts.length > 3 && <div className="text-[10px] text-slate-400 font-bold">+{alerts.length - 3}</div>}
                </div>
              </div>
            </div>
          </div>

          <div className="md:col-span-1 lg:col-span-2 bg-white p-6 rounded-3xl shadow-sm border border-slate-100 text-slate-900 relative overflow-hidden group">
            <img 
              src="https://images.unsplash.com/photo-1554469384-e58fac16e23a?auto=format&fit=crop&q=80&w=1000" 
              alt="Bourse de Casablanca Building" 
              className="absolute inset-0 w-full h-full object-cover opacity-10 group-hover:scale-110 transition-transform duration-1000"
              referrerPolicy="no-referrer"
            />
            <div className="relative z-10">
              <h2 className="text-xl font-bold mb-2 text-blue-600">Analyse IA du Marché</h2>
              <p className="text-slate-600 text-sm leading-relaxed max-w-md font-medium">
                Le marché montre une résilience notable dans le secteur bancaire. Les investisseurs privilégient les valeurs de rendement en prévision des résultats annuels.
              </p>
              <div className="mt-4 flex gap-2">
                <span className="px-2 py-1 bg-blue-50 text-blue-600 rounded-lg text-[10px] font-bold uppercase border border-blue-100">Tendance: Neutre</span>
                <span className="px-2 py-1 bg-blue-50 text-blue-600 rounded-lg text-[10px] font-bold uppercase border border-blue-100">Volume: Élevé</span>
              </div>
            </div>
            <div className="absolute right-0 bottom-0 opacity-5 transform translate-x-4 translate-y-4 group-hover:scale-110 transition-transform duration-700">
              <Activity className="w-48 h-48 text-blue-600" />
            </div>
          </div>
        </section>

        {/* Sectors */}
        <section className="overflow-x-auto pb-4 -mx-6 px-6 scrollbar-hide">
          <div className="flex gap-4 min-w-max">
            {sectors.map((sector) => (
              <div key={sector.name} className="bg-white px-4 py-3 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-3">
                <span className="text-xl">{sector.icon}</span>
                <div>
                  <p className="text-xs font-bold text-slate-800">{sector.name}</p>
                  <p className={cn(
                    "text-[10px] font-bold",
                    sector.change >= 0 ? "text-emerald-500" : "text-rose-500"
                  )}>
                    {sector.change >= 0 ? '+' : ''}{sector.change}%
                  </p>
                </div>
              </div>
            ))}
          </div>
        </section>

        {/* Top Gainers & Flops */}
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          {/* Top Gainers */}
          <section>
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-lg font-bold flex items-center gap-2">
                <TrendingUp className="w-5 h-5 text-emerald-500" />
                Top Gainers
              </h2>
            </div>
            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
              {topGainers.map((stock, idx) => (
                <motion.div 
                  key={stock.symbol}
                  initial={{ opacity: 0, y: 20 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ delay: idx * 0.1 }}
                  onClick={() => setSelectedStock(stock)}
                  className="bg-white p-4 rounded-3xl shadow-sm border border-slate-100 hover:border-emerald-200 hover:shadow-md transition-all cursor-pointer group"
                >
                  <div className="flex justify-between items-start mb-3">
                    <div className="w-10 h-10 rounded-xl overflow-hidden border border-slate-100 group-hover:border-emerald-100 transition-colors">
                      <img 
                        src={stock.logo || `https://picsum.photos/seed/${stock.symbol}/100/100`} 
                        alt={stock.name} 
                        className="w-full h-full object-cover"
                        referrerPolicy="no-referrer"
                      />
                    </div>
                    <span className="px-2 py-1 rounded-lg text-[10px] font-bold bg-emerald-50 text-emerald-600">
                      +{stock.changePercent.toFixed(2)}%
                    </span>
                  </div>
                  <h3 className="font-bold text-sm text-slate-800 truncate">{stock.name}</h3>
                  <div className="flex items-end justify-between mt-2">
                    <span className="text-md font-bold">{stock.price.toLocaleString()} <span className="text-[10px] font-normal text-slate-400">MAD</span></span>
                    <div className="h-6 w-12">
                      <ResponsiveContainer width="100%" height="100%">
                        <LineChart data={stock.history}>
                          <Line type="monotone" dataKey="price" stroke="#10b981" strokeWidth={2} dot={false} />
                        </LineChart>
                      </ResponsiveContainer>
                    </div>
                  </div>
                </motion.div>
              ))}
            </div>
          </section>

          {/* Top Flops */}
          <section>
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-lg font-bold flex items-center gap-2">
                <TrendingDown className="w-5 h-5 text-rose-500" />
                Top Flops
              </h2>
            </div>
            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
              {topFlops.map((stock, idx) => (
                <motion.div 
                  key={stock.symbol}
                  initial={{ opacity: 0, y: 20 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ delay: idx * 0.1 }}
                  onClick={() => setSelectedStock(stock)}
                  className="bg-white p-4 rounded-3xl shadow-sm border border-slate-100 hover:border-rose-200 hover:shadow-md transition-all cursor-pointer group"
                >
                  <div className="flex justify-between items-start mb-3">
                    <div className="w-10 h-10 rounded-xl overflow-hidden border border-slate-100 group-hover:border-rose-100 transition-colors">
                      <img 
                        src={stock.logo || `https://picsum.photos/seed/${stock.symbol}/100/100`} 
                        alt={stock.name} 
                        className="w-full h-full object-cover"
                        referrerPolicy="no-referrer"
                      />
                    </div>
                    <span className="px-2 py-1 rounded-lg text-[10px] font-bold bg-rose-50 text-rose-600">
                      {stock.changePercent.toFixed(2)}%
                    </span>
                  </div>
                  <h3 className="font-bold text-sm text-slate-800 truncate">{stock.name}</h3>
                  <div className="flex items-end justify-between mt-2">
                    <span className="text-md font-bold">{stock.price.toLocaleString()} <span className="text-[10px] font-normal text-slate-400">MAD</span></span>
                    <div className="h-6 w-12">
                      <ResponsiveContainer width="100%" height="100%">
                        <LineChart data={stock.history}>
                          <Line type="monotone" dataKey="price" stroke="#f43f5e" strokeWidth={2} dot={false} />
                        </LineChart>
                      </ResponsiveContainer>
                    </div>
                  </div>
                </motion.div>
              ))}
            </div>
          </section>
        </div>

        {/* Main List & Details */}
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
          {/* Stock List */}
          <section className="lg:col-span-2 space-y-4">
            <div className="flex items-center justify-between">
              <h2 className="text-lg font-bold">Toutes les Sociétés</h2>
              <span className="text-xs font-bold text-slate-400 uppercase">{filteredStocks.length} Valeurs</span>
            </div>
            <div className="bg-white rounded-3xl shadow-sm border border-slate-100 overflow-hidden">
              <div className="overflow-x-auto">
                <table className="w-full text-left border-collapse">
                  <thead>
                    <tr className="border-b border-slate-50">
                      <th className="px-6 py-4 text-xs font-bold text-slate-400 uppercase tracking-wider">Société</th>
                      <th className="px-6 py-4 text-xs font-bold text-slate-400 uppercase tracking-wider">Prix (MAD)</th>
                      <th className="px-6 py-4 text-xs font-bold text-slate-400 uppercase tracking-wider">Variation</th>
                      <th className="px-6 py-4 text-xs font-bold text-slate-400 uppercase tracking-wider">Volume</th>
                      <th className="px-6 py-4"></th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-slate-50">
                    {filteredStocks.map((stock) => (
                      <tr 
                        key={stock.symbol} 
                        onClick={() => setSelectedStock(stock)}
                        className="hover:bg-slate-50 cursor-pointer transition-colors group"
                      >
                        <td className="px-6 py-4">
                          <div className="flex items-center gap-3">
                            <div className="w-10 h-10 rounded-xl bg-slate-100 flex items-center justify-center overflow-hidden border border-slate-50 group-hover:border-blue-100 transition-colors">
                              {stock.logo ? (
                                <img 
                                  src={stock.logo} 
                                  alt={stock.name} 
                                  className="w-full h-full object-cover"
                                  referrerPolicy="no-referrer"
                                />
                              ) : (
                                <span className="text-xs font-bold text-slate-400">{stock.symbol.substring(0, 2)}</span>
                              )}
                            </div>
                            <div>
                              <p className="font-bold text-sm text-slate-800">{stock.name}</p>
                              <p className="text-[10px] text-slate-400 font-mono">{stock.symbol}</p>
                            </div>
                          </div>
                        </td>
                        <td className="px-6 py-4 font-mono text-sm font-bold">
                          {stock.price.toLocaleString()}
                        </td>
                        <td className="px-6 py-4">
                          <div className={cn(
                            "flex items-center gap-1 text-sm font-bold",
                            stock.changePercent >= 0 ? "text-emerald-600" : "text-rose-600"
                          )}>
                            {stock.changePercent >= 0 ? <TrendingUp className="w-4 h-4" /> : <TrendingDown className="w-4 h-4" />}
                            {Math.abs(stock.changePercent).toFixed(2)}%
                          </div>
                        </td>
                        <td className="px-6 py-4 text-sm text-slate-500">
                          {stock.volume.toLocaleString()}
                        </td>
                        <td className="px-6 py-4 text-right">
                          <button className="p-2 hover:bg-white rounded-full transition-colors">
                            <ArrowUpRight className="w-4 h-4 text-slate-300 group-hover:text-blue-500" />
                          </button>
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
            </div>
          </section>

          {/* Details Sidebar */}
          <aside className="space-y-6">
            <AnimatePresence mode="wait">
              {selectedStock ? (
                <motion.div 
                  key={selectedStock.symbol}
                  initial={{ opacity: 0, x: 20 }}
                  animate={{ opacity: 1, x: 0 }}
                  exit={{ opacity: 0, x: 20 }}
                  className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100 sticky top-28"
                >
                  <div className="flex justify-between items-start mb-6">
                    <div className="flex items-center gap-3">
                      <div className="w-12 h-12 rounded-2xl overflow-hidden border border-slate-100">
                        <img 
                          src={selectedStock.logo || `https://picsum.photos/seed/${selectedStock.symbol}/100/100`} 
                          alt={selectedStock.name} 
                          className="w-full h-full object-cover"
                          referrerPolicy="no-referrer"
                        />
                      </div>
                      <div>
                        <h2 className="text-xl font-bold">{selectedStock.name}</h2>
                        <p className="text-sm text-slate-400 font-mono">{selectedStock.symbol} • {selectedStock.sector}</p>
                      </div>
                    </div>
                    <button 
                      onClick={() => setSelectedStock(null)}
                      className="text-slate-400 hover:text-slate-600"
                    >
                      ×
                    </button>
                  </div>

                  <div className="grid grid-cols-2 gap-4 mb-8">
                    <div className="bg-slate-50 p-4 rounded-2xl">
                      <p className="text-[10px] text-slate-400 uppercase font-bold mb-1">Prix Actuel</p>
                      <p className="text-lg font-bold">{selectedStock.price.toLocaleString()} <span className="text-xs font-normal">MAD</span></p>
                    </div>
                    <div className="bg-slate-50 p-4 rounded-2xl">
                      <p className="text-[10px] text-slate-400 uppercase font-bold mb-1">Cap. Boursière</p>
                      <p className="text-lg font-bold">{selectedStock.marketCap}</p>
                    </div>
                  </div>

                  <div className="h-48 mb-6">
                    <ResponsiveContainer width="100%" height="100%">
                      <AreaChart data={selectedStock.history}>
                        <defs>
                          <linearGradient id="colorPrice" x1="0" y1="0" x2="0" y2="1">
                            <stop offset="5%" stopColor="#2563eb" stopOpacity={0.1}/>
                            <stop offset="95%" stopColor="#2563eb" stopOpacity={0}/>
                          </linearGradient>
                        </defs>
                        <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />
                        <XAxis dataKey="date" hide />
                        <YAxis hide domain={['auto', 'auto']} />
                        <Tooltip 
                          contentStyle={{ borderRadius: '12px', border: 'none', boxShadow: '0 4px 12px rgba(0,0,0,0.05)' }}
                          labelStyle={{ display: 'none' }}
                        />
                        <Area 
                          type="monotone" 
                          dataKey="price" 
                          stroke="#2563eb" 
                          strokeWidth={3}
                          fillOpacity={1} 
                          fill="url(#colorPrice)" 
                        />
                      </AreaChart>
                    </ResponsiveContainer>
                  </div>

                  <div className="space-y-4">
                    <div className="bg-slate-50 p-4 rounded-2xl border border-slate-100">
                      <h3 className="text-xs font-bold uppercase tracking-wider text-slate-500 mb-3 flex items-center gap-2">
                        <Bell className="w-3 h-3" />
                        Nouvelle Alerte de Prix
                      </h3>
                      <div className="flex flex-col gap-3">
                        <div className="flex gap-2">
                          <select 
                            value={newAlertCondition}
                            onChange={(e) => setNewAlertCondition(e.target.value as 'above' | 'below')}
                            className="bg-white border border-slate-200 rounded-xl px-3 py-2 text-xs font-bold focus:outline-none focus:ring-2 focus:ring-blue-500/20"
                          >
                            <option value="above">Prix ≥</option>
                            <option value="below">Prix ≤</option>
                          </select>
                          <input 
                            type="number" 
                            placeholder="Prix cible (MAD)"
                            value={newAlertPrice}
                            onChange={(e) => setNewAlertPrice(e.target.value)}
                            className="flex-1 bg-white border border-slate-200 rounded-xl px-3 py-2 text-xs font-mono focus:outline-none focus:ring-2 focus:ring-blue-500/20"
                          />
                        </div>
                        <button 
                          onClick={addAlert}
                          className="w-full py-2 bg-blue-600 text-white rounded-xl font-bold text-xs hover:bg-blue-700 transition-colors flex items-center justify-center gap-2"
                        >
                          <Plus className="w-3 h-3" />
                          Ajouter l'alerte
                        </button>
                      </div>
                    </div>

                    <div className="flex items-center gap-2 text-blue-600 bg-blue-50 p-3 rounded-2xl text-xs font-bold">
                      <Calendar className="w-4 h-4" />
                      Données du {lastUpdated.toLocaleDateString('fr-FR', { day: 'numeric', month: 'long', year: 'numeric' })}
                    </div>
                    <div className="flex items-center gap-2 text-slate-600 bg-slate-50 p-3 rounded-2xl text-xs font-bold">
                      <Info className="w-4 h-4" />
                      Analyse de Performance
                    </div>
                    <p className="text-sm text-slate-600 leading-relaxed">
                      {selectedStock.name} présente une tendance {selectedStock.changePercent > 0 ? 'haussière' : 'baissière'} sur les dernières sessions. Le volume d'échange de {selectedStock.volume.toLocaleString()} titres indique un intérêt {selectedStock.volume > 10000 ? 'soutenu' : 'modéré'} du marché.
                    </p>
                    <button className="w-full py-3 bg-slate-900 text-white rounded-2xl font-bold text-sm hover:bg-slate-800 transition-colors">
                      Voir l'analyse complète
                    </button>
                  </div>
                </motion.div>
              ) : (
                <div className="space-y-6 sticky top-28">
                  <div className="bg-slate-100/50 border-2 border-dashed border-slate-200 p-12 rounded-3xl text-center">
                    <div className="bg-white w-12 h-12 rounded-2xl shadow-sm flex items-center justify-center mx-auto mb-4">
                      <PieChart className="w-6 h-6 text-slate-300" />
                    </div>
                    <h3 className="font-bold text-slate-500 mb-2">Sélectionnez une valeur</h3>
                    <p className="text-xs text-slate-400">Cliquez sur une société pour voir ses détails financiers et son graphique.</p>
                  </div>

                  <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100">
                    <h3 className="text-sm font-bold uppercase tracking-wider text-slate-400 mb-4 flex items-center gap-2">
                      <BellRing className="w-4 h-4 text-amber-500" />
                      Mes Alertes
                    </h3>
                    <div className="space-y-3">
                      {alerts.length > 0 ? (
                        alerts.map(alert => {
                          const currentStock = stocks.find(s => s.symbol === alert.symbol);
                          const isTriggered = currentStock && (
                            alert.condition === 'above' 
                              ? currentStock.price >= alert.targetPrice 
                              : currentStock.price <= alert.targetPrice
                          );
                          
                          return (
                            <div key={alert.id} className={cn(
                              "p-3 rounded-2xl border transition-all",
                              isTriggered ? "bg-amber-50 border-amber-200" : "bg-slate-50 border-slate-100"
                            )}>
                              <div className="flex justify-between items-start mb-1">
                                <span className="text-xs font-bold text-slate-800">{alert.symbol}</span>
                                <button 
                                  onClick={() => removeAlert(alert.id)}
                                  className="text-slate-300 hover:text-rose-500 transition-colors"
                                >
                                  <Trash2 className="w-3 h-3" />
                                </button>
                              </div>
                              <div className="flex justify-between items-end">
                                <p className="text-[10px] text-slate-500">
                                  {alert.condition === 'above' ? 'Supérieur à' : 'Inférieur à'} <span className="font-bold">{alert.targetPrice} MAD</span>
                                </p>
                                {isTriggered && (
                                  <span className="text-[10px] font-bold text-amber-600 animate-pulse">Déclenchée !</span>
                                )}
                              </div>
                            </div>
                          );
                        })
                      ) : (
                        <p className="text-xs text-slate-400 italic">Aucune alerte définie.</p>
                      )}
                    </div>
                  </div>

                  <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100">
                    <h3 className="text-sm font-bold uppercase tracking-wider text-slate-400 mb-4 flex items-center justify-between">
                      <div className="flex items-center gap-2">
                        <Globe className="w-4 h-4" />
                        Actualités du Marché
                      </div>
                      {loadingNews && <RefreshCw className="w-3 h-3 animate-spin" />}
                    </h3>
                    <div className="space-y-6">
                      {news.map((item, i) => (
                        <div key={i} className="group cursor-pointer">
                          <div className="flex items-center gap-2 mb-1">
                            <span className={cn(
                              "px-1.5 py-0.5 rounded text-[8px] font-bold uppercase",
                              item.sentiment === 'positive' ? "bg-emerald-100 text-emerald-600" :
                              item.sentiment === 'negative' ? "bg-rose-100 text-rose-600" :
                              "bg-slate-100 text-slate-600"
                            )}>
                              {item.sentiment === 'positive' ? 'Positif' : 
                               item.sentiment === 'negative' ? 'Négatif' : 'Neutre'}
                            </span>
                            <p className="text-[10px] text-slate-400">{item.time}</p>
                          </div>
                          <p className="text-sm font-bold text-slate-700 group-hover:text-blue-600 transition-colors line-clamp-2">{item.title}</p>
                          <p className="text-xs text-slate-500 mt-1 leading-relaxed italic">"{item.summary}"</p>
                        </div>
                      ))}
                    </div>
                  </div>
                </div>
              )}
            </AnimatePresence>
          </aside>
        </div>
      </main>

      {/* Footer with Bright Bourse de Casablanca Vibe */}
      <footer className="relative mt-24 overflow-hidden rounded-t-[60px] bg-white text-slate-900 border-t border-slate-100">
        <img 
          src="https://images.unsplash.com/photo-1560179707-f14e90ef3623?auto=format&fit=crop&q=80&w=2000" 
          alt="Bourse de Casablanca" 
          className="absolute inset-0 w-full h-full object-cover opacity-5"
          referrerPolicy="no-referrer"
        />
        <div className="relative z-10 max-w-7xl mx-auto px-12 py-24">
          <div className="grid grid-cols-1 md:grid-cols-3 gap-16">
            <div className="space-y-6">
              <div className="flex items-center gap-3">
                <div className="bg-blue-600 p-2 rounded-xl">
                  <Activity className="text-white w-6 h-6" />
                </div>
                <h2 className="text-2xl font-black tracking-tighter">CasaFinance</h2>
              </div>
              <p className="text-slate-500 text-sm leading-relaxed font-medium">
                Votre partenaire de confiance pour l'analyse boursière au Maroc. Données en direct, alertes intelligentes et analyse par IA.
              </p>
            </div>
            <div className="space-y-6">
              <h3 className="text-xs font-black uppercase tracking-widest text-blue-600">Navigation</h3>
              <ul className="space-y-4 text-sm font-bold text-slate-600">
                <li><a href="#" className="hover:text-blue-600 transition-colors">Tableau de Bord</a></li>
                <li><a href="#" className="hover:text-blue-600 transition-colors">Alertes de Prix</a></li>
                <li><a href="#" className="hover:text-blue-600 transition-colors">Actualités</a></li>
                <li><a href="#" className="hover:text-blue-600 transition-colors">Exportation</a></li>
              </ul>
            </div>
            <div className="space-y-6">
              <h3 className="text-xs font-black uppercase tracking-widest text-blue-600">Légal</h3>
              <ul className="space-y-4 text-sm font-bold text-slate-600">
                <li><a href="#" className="hover:text-blue-600 transition-colors">Mentions Légales</a></li>
                <li><a href="#" className="hover:text-blue-600 transition-colors">Confidentialité</a></li>
                <li><a href="#" className="hover:text-blue-600 transition-colors">Conditions d'Utilisation</a></li>
              </ul>
            </div>
          </div>
          <div className="mt-24 pt-12 border-t border-slate-100 flex flex-col md:flex-row justify-between items-center gap-6 text-slate-400 text-[10px] font-black uppercase tracking-widest">
            <span>© 2026 CasaFinance. Tous droits réservés.</span>
            <div className="flex gap-8">
              <span>Bourse de Casablanca</span>
              <span>Casa Finance City</span>
            </div>
          </div>
        </div>
      </footer>
    </div>
  );
